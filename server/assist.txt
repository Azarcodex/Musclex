import Product from "../../models/products/product.js";
import Variant from "../../models/products/variant.js";

export const getAllProducts = async (req, res) => {
  try {
    // 1ï¸âƒ£ Get all active, non-deleted products
    const products = await Product.find({ isDeleted: false, status: "Active" })
      .populate("brandID", "name")
      .populate("catgid", "catgName")
      .lean();

    // 2ï¸âƒ£ Get all variants for these products
    const productIds = products.map((p) => p._id);
    const variants = await Variant.find({
      productId: { $in: productIds },
      isListed: true,
    }).lean();

    // 3ï¸âƒ£ Merge variant data with each product
    const result = products.map((prod) => {
      const prodVariants = variants.filter(
        (v) => v.productId.toString() === prod._id.toString()
      );

      // Extract sale prices
      const prices = prodVariants.map((v) => v.salePrice);
      const minPrice = prices.length ? Math.min(...prices) : null;
      const maxPrice = prices.length ? Math.max(...prices) : null;

      // Extract first variant image (or default)
      const images = prodVariants.flatMap((v) => v.images || []);
      const previewImage = images.length > 0 ? images[0] : "/placeholder.png";

      return {
        ...prod,
        priceSummary: { minPrice, maxPrice },
        previewImage, // ðŸ‘ˆ use this in frontend
        variants: prodVariants,
      };
    });

    return res.status(200).json({
      success: true,
      data: result,
    });
  } catch (error) {
    console.error(error);
    return res.status(500).json({
      success: false,
      message: "Failed to fetch products",
      error: error.message,
    });
  }
};
